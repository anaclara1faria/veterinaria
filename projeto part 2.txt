CREATE DATABASE farmacia;

CREATE TABLE Tipo_De_Produto(idTipo_De_Produto INT PRIMARY KEY AUTO_INCREMENT, nome VARCHAR(45));

CREATE TABLE endereco(idendereco INT PRIMARY KEY AUTO_INCREMENT, rua VARCHAR(45),numero INT,bairro VARCHAR(45),cidade VARCHAR(45),estado VARCHAR(45));

CREATE TABLE Produto(idProduto INT PRIMARY KEY AUTO_INCREMENT, codigo INT, designacao VARCHAR(45), composicao VARCHAR(45), fabricante VARCHAR(45), preco INT);

CREATE TABLE venda(idvenda INT PRIMARY KEY AUTO_INCREMENT, codigo_de_cliente VARCHAR(45),produtos_adquiridos VARCHAR(45));

CREATE TABLE Produto(idcliente INT PRIMARY KEY AUTO_INCREMENT, nome VARCHAR(45), telefone INT, codigo_postal VARCHAR(45), numero_do _contribuinte VARCHAR(45));
